# Step 1: Setup your machine with Git Configuration
git --version
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"
git config --list

# Step 2: Integrate notepad++.exe to Git and make it a default editor
notepad++
alias notepad++="'C:/Program Files/Notepad++/notepad++.exe'"
git config --global core.editor "'C:/Program Files/Notepad++/notepad++.exe' -multiInst -notabbar -nosession -noPlugin"
git config --global -e

# Step 3: Add a file to source code repository
mkdir GitDemo
cd GitDemo
git init
ls -a
echo "Welcome to Git" > welcome.txt
ls
cat welcome.txt
git status
git add welcome.txt
# This will open notepad++ for the commit message
git commit 
git status
git remote add origin <REMOTE_URL>
git pull origin master
git push origin master
